115  void units(std::vector<std::string> &units) {
116  std::vector<std::string> sanitized;
117  sanitized.reserve(units.size());
118  std::transform(begin(units), end(units), std::back_inserter(sanitized), [](const std::string &x) {
119  std::string unit = util::unitSanitizer(x);
120  if (unit.length() > 0 && (unit != "none" && !(util::isSIUnit(unit)))) {
121  std::string msg = "Unit " + unit +" is not a SI unit. Note: so far only atomic SI units are supported.";
122  throw InvalidUnit(msg, "SimpleTag::units(vector<string> &units)");
123  }
124  return unit;
125  });
126  backend()->units(sanitized);
127  }
