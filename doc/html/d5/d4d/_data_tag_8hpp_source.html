151  void units(std::vector<std::string> &units) {
152  std::vector<std::string> sanitized;
153  sanitized.reserve(units.size());
154  std::transform(begin(units), end(units), std::back_inserter(sanitized), [](const std::string &x) {
155  std::string unit = util::unitSanitizer(x);
156  if (unit.length() > 0 && (unit != "none" && !(util::isSIUnit(unit)))) {
157  std::string msg = "Unit " + unit +" is not a SI unit. Note: so far only atomic SI units are supported.";
158  throw InvalidUnit(msg, "DataTag::units(vector<string> &units)");
159  }
160  return unit;
161  });
162  backend()->units(sanitized);
163  }
